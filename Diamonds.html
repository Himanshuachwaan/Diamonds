# Interquartile range Outlier Detection
def cal_iqr_outlier(df,col):
    ser =df[col]
    iqr = ser.quantile(.75) - ser.quantile(.25)
    med = ser.median()
    small_mask = ser < med-iqr*3 # small values are outliers
    large_mask = ser > med+iqr*3 # large values are outliers
    return small_mask | large_mask # combine mask
